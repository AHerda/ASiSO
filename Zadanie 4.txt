Start,	Clear					/ Start
		Input
		Skipcond	400
		Jump		Start2
		Jump		WypiszZero

Start2,	Store		Liczba
		Skipcond	800
		Subt		Liczba
		Skipcond	800
		Subt		Liczba
		Store		Liczba
		Store		LiczbaTemp

While1,	Load		LiczbaTemp		/ while 1 condition
		Skipcond	800
		Jump		Output_R		/ do dokonczenia, skip do konca petli
		
While1_Do,	Load		Licznik
		Add		ONE
		Store		Licznik
		Load		LiczbaTemp
		Jump		Dzielenie


Output_R,	Jump		Wypisz

Checker1,	Clear
		Add		ONE
		Store		Checker

		
While2,	Load		Licznik		/ While2 condition
		Skipcond	800
		Jump		Koniec		/ do dokonczenia, skip do konca petli

While2_do,	Store		Licznik2
		Load		Liczba
		Store		LiczbaTemp

WhileM,	Load		Licznik2
		Skipcond	800
		Jump		While2_do2		/ do dokonczenia, skip do konca petli
		
WhileM_do,	Subt		ONE
		Store		Licznik2
		Load		LiczbaTemp
		Jump		Dzielenie
		
While2_do2,	Load		Licznik
		Subt		ONE
		Store		Licznik
		Jump		Wypisz



		
Koniec,	Clear
		Halt
		

WypiszZero,	Clear
		AddI 		str_ptr
		Output
		Jump		Koniec



		
		
		
Dzielenie,	Store		Dzielna
Dzielenie2,	Subt		SIXTEEN
		Skipcond	000
		Jump		Dzielenie3
		Load		Dzielna
		Store		Reszta
		Load		Wynik_d
		Store		LiczbaTemp
		Clear
		Store		Wynik_d
		Load		Checker
		Skipcond	800
		Jump		While1
		Jump		WhileM

Dzielenie3,	Store		Dzielna
		Load		Wynik_d
		Add		ONE
		Store		Wynik_d
		Load		Dzielna
		Jump		Dzielenie2

Wypisz,	Load		str_ptr
		Add		Reszta
		Store		Temp_ptr
		Clear
		AddI		Temp_ptr
		Output


WypiszEnd,	Clear
		Store		Reszta
		Load		Checker
		Skipcond	800
		Jump		Checker1
		Jump		While2

str_ptr,	HEX		50			/ pointer to the start of the string
str,		HEX		30			/ 0
		HEX		31			/ 1
		HEX		32			/ 2
		HEX		33			/ 3
		HEX		34			/ 4
		HEX		35			/ 5
		HEX		36			/ 6
		HEX		37			/ 7
		HEX		38			/ 8
		HEX		39			/ 9
		HEX		41			/ A = 10
		HEX		42			/ B = 11
		HEX		43			/ C = 12
		HEX		44			/ D = 13
		HEX		45			/ E = 14
		HEX		46			/ F = 15
			

/ zmienne
Liczba,	DEC		0
LiczbaTemp,	DEC		0
Licznik,	DEC		-1
Licznik2,	DEC		0
Reszta,	DEC		0
Checker,	DEC		0
Temp_ptr,	DEC		0





/ zmienne dzielenie
Wynik_d,	DEC		0
Dzielna,	DEC		0




/ stale
ZERO,		DEC		0
ONE,		DEC		1
SIXTEEN,	DEC		16
